-------renombrar ramas GIT ------
$ git branch --set-upstream-to=origin/feature-caratulas arquitectura


---------comandos de GIT--------
------clonar repositorio--------
git clone <url_repo>
----antes de todo descargar los cambios de la rama
git pull origin <rama>

------listar las ramas-------
git branch
git branch -a
------selecionar rama---------
git checkout <rama>
-------listar el estdo-----
git status
-------agregar los ficheros -------
git add <ficheros>
-------agregar el commit-------
git commit -m "Initial commit"
git status
--------subir los cambios--------
git push -u origin <rama>


------- otros comandos Git----------
----desacer el commit----
git reset --hard HEAD~1
-----guardar el estado de los documentos temporalmente------
git stash
-----recuperar el estado de los documentos------
git stash apply

Git global setup
git config --global user.name "Erick Saul Gonzalez Gonzalez"

Create a new repository
git clone http://git.gnp.mx/GCP/gnp-plus/gnp-plus-portal/pruebas-gtec.git
cd pruebas-gtec
touch README.md
git add README.md
git commit -m "add README"
git push -u origin master

Push an existing folder
cd existing_folder
git init
git remote add origin http://git.gnp.mx/GCP/gnp-plus/gnp-plus-portal/pruebas-gtec.git
git add .
git commit -m "Initial commit"
git push -u origin master

Push an existing Git repository
cd existing_repo
git remote rename origin old-origin
git remote add origin http://git.gnp.mx/GCP/gnp-plus/gnp-plus-portal/pruebas-gtec.git
git remote add origin http://git.gnp.mx/GCP/gnp-plus/gnp-plus-front/PLUS-ANGULAR-FRONT.git

git push -u origin --all
git push -u origin --tags
